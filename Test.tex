    \begin{paragraph}{1.2)}
        \textit{
            Evaluate the following definite integral:
        }

        \begin{equation*}
            \int_{0}^{3} \frac{2x}{\sqrt{x^2+4}} dx 
        \end{equation*}
        
        $$
            u = x^2 + 4 
            \quad
            du = 2xdx
        $$
        
        \begin{equation*}
            \Rightarrow\quad
            \int_{0^2+4}^{3^2+4} \frac{1}{\sqrt{u}} du = 
            \left. 2\sqrt{u} \right |_{4}^{13}
        \end{equation*}
        
        \begin{equation*}
            2\left(\sqrt{13} - \sqrt{4}\right) \approx \framebox{3.211}
        \end{equation*}
    \end{paragraph}
